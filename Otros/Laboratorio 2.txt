import sqlite3
from datetime import datetime

class DatabaseManager:
    """Clase encargada de la persistencia de datos"""

    def __init__(self, db_name="sentiment_iq.db"):
        self.db_name = db_name
        self.init_db()
    
    def init_db(self):
        """Crea la tabla"""
        conn = sqlite3.connect(self.db_name)
        cursor = conn.cursor()
        cursor.execute(''' 
            CREATE TABLE IF NOT EXISTS reviews (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                producto TEXT NOT NULL,
                comentario TEXT NOT NULL,
                sentimiento_ia TEXT,
                fecha TEXT
            )
        
        ''')
        conn.commit()
        conn.close()
        print("Base de datos inicializada correctamente.")

    def registrar_reviews(self, producto, comentario, sentimiento="Pendiente"):
        """Insertar una nueva reseña en la base de datos"""
        fecha_actual = datetime.now().strftime("d/m/Y, H:MS")

        try:
            conn = sqlite3.connect(self.db_name)
            cursor = conn.cursor()
            cursor.execute('''
                INSERT INTO reviews (producto, comentario, sentimiento_ia, fecha)
                VALUES (?, ?, ?, ?)            
            ''' , (producto, comentario, sentimiento, fecha_actual))
            conn.commit()
            print(f"Éxito: Reseña de '{producto}' guardada.")
                    
        except Exception as e:
            print(f"Error al guardar {e}")
        finally:
            conn.close()
